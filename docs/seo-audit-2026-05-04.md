## Phase 3 — Schema.org (kod gotowy do wklejenia)

### TOP 3 quick wins (~25 min, 0 zewnętrznych zależności)

**F. Block Article z pages** — RankMath generuje błędny Article schema na pages:
```php
add_filter( 'rank_math/snippet/rich_snippet_article_entity', 'gorvita_block_article_on_pages', 10, 1 );
function gorvita_block_article_on_pages( $entity ) {
    if ( is_singular( 'page' ) || is_front_page() || is_home() ) {
        return false;
    }
    return $entity;
}
```

**B. Brand + manufacturer w Product** — wymóg GMC, naprawia 106 produktów naraz:
```php
add_filter( 'rank_math/snippet/rich_snippet_product_entity', 'gorvita_add_brand_to_product', 10, 1 );
function gorvita_add_brand_to_product( $entity ) {
    $entity['brand'] = array(
        '@type' => 'Brand',
        'name'  => 'Gorvita',
        'logo'  => get_stylesheet_directory_uri() . '/assets/images/logo.svg',
    );
    $entity['manufacturer'] = array(
        '@type' => 'Organization',
        'name'  => 'PPUH Gorvita Sp. z o.o.',
        'url'   => home_url( '/' ),
    );
    return $entity;
}
```

**C. hasMerchantReturnPolicy + shippingDetails** — wymóg GMC od 2025:
```php
add_filter( 'rank_math/snippet/rich_snippet_product_entity', 'gorvita_add_offer_policies', 20, 1 );
function gorvita_add_offer_policies( $entity ) {
    if ( empty( $entity['offers'] ) ) return $entity;
    $return_policy = array(
        '@type'                => 'MerchantReturnPolicy',
        'applicableCountry'    => 'PL',
        'returnPolicyCategory' => 'https://schema.org/MerchantReturnFiniteReturnWindow',
        'merchantReturnDays'   => 14,
        'returnMethod'         => 'https://schema.org/ReturnByMail',
        'returnFees'           => 'https://schema.org/FreeReturn',
    );
    $shipping = array(
        '@type'        => 'OfferShippingDetails',
        'shippingRate' => array( '@type' => 'MonetaryAmount', 'value' => '0', 'currency' => 'PLN' ),
        'shippingDestination' => array( '@type' => 'DefinedRegion', 'addressCountry' => 'PL' ),
        'deliveryTime' => array(
            '@type'        => 'ShippingDeliveryTime',
            'handlingTime' => array( '@type' => 'QuantitativeValue', 'minValue' => 0, 'maxValue' => 1, 'unitCode' => 'DAY' ),
            'transitTime'  => array( '@type' => 'QuantitativeValue', 'minValue' => 1, 'maxValue' => 2, 'unitCode' => 'DAY' ),
        ),
    );
    if ( isset( $entity['offers']['@type'] ) && $entity['offers']['@type'] === 'Offer' ) {
        $entity['offers']['hasMerchantReturnPolicy'] = $return_policy;
        $entity['offers']['shippingDetails']         = $shipping;
    }
    return $entity;
}
```

### D. FAQ shortcode

```php
add_shortcode( 'gorvita_faq', 'gorvita_faq_shortcode' );
function gorvita_faq_shortcode( $atts, $content = '' ) {
    $items = json_decode( '[' . trim( wp_strip_all_tags( $content ) ) . ']', true );
    if ( ! is_array( $items ) ) return '';
    $html = '<section class="gorvita-faq"><h2>Najczęściej zadawane pytania</h2><dl>';
    $main = array();
    foreach ( $items as $it ) {
        $q = esc_html( $it['q'] ?? '' ); $a = esc_html( $it['a'] ?? '' );
        $html .= "<dt>{$q}</dt><dd>{$a}</dd>";
        $main[] = array( '@type' => 'Question', 'name' => $q, 'acceptedAnswer' => array( '@type' => 'Answer', 'text' => $a ) );
    }
    $html .= '</dl></section>';
    $schema = array( '@context' => 'https://schema.org', '@type' => 'FAQPage', 'mainEntity' => $main );
    $html  .= '<script type="application/ld+json">' . wp_json_encode( $schema, JSON_UNESCAPED_UNICODE | JSON_UNESCAPED_SLASHES ) . '</script>';
    return $html;
}
```
